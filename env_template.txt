# Enhanced Fake News Detection System - API Configuration
# Copy this file to .env and add your actual API keys

# Gemini AI API Key (Required for AI-powered fact checking)
# Get from: https://makersuite.google.com/app/apikey
GEMINI_API_KEY=YOUR_GEMINI_API_KEY_HERE

# SerpAPI Key (Recommended for best search results)
# Get from: https://serpapi.com/dashboard
SERPAPI_KEY=YOUR_SERPAPI_KEY_HERE

# Google Custom Search API (Alternative to SerpAPI)
# Get from: https://developers.google.com/custom-search/v1/introduction
GOOGLE_SEARCH_API_KEY=YOUR_GOOGLE_SEARCH_API_KEY_HERE
GOOGLE_CSE_ID=YOUR_GOOGLE_CSE_ID_HERE

# Flask Configuration
FLASK_DEBUG=true
PORT=5000

# Notes:
# - GEMINI_API_KEY is essential for AI-powered fact checking
# - At least one search API (SERPAPI_KEY or GOOGLE_SEARCH_API_KEY) is needed for real-time verification
# - SerpAPI is easier to set up and provides better results
# - Without these APIs, the system will work with basic ML classification only
